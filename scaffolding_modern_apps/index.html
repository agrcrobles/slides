app.factory("Pin", function ($resource, ENV) {
  return $resource(ENV.apiEndpoint + "/pins/:id.json", {id: "@id"},
    {
      'create': {method: 'POST'},
      'index': {method: 'GET', isArray: true},
      'show': {method: 'GET', isArray: false},
      'update': {method: 'PUT'},
      'destroy': {method: 'DELETE'}
    }
  );
});
